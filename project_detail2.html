System Utility Development

Project

Using Python to obtain invaluable metrics on system CPU, RAM, disk, SSD, and network.

Project Goals

The primary objective of this project was to develop a tool that analyzes crucial hardware metrics, providing the user with valuable data about hardware health and efficiency.

Challenges Faced

One of the main challenges during this project was ensuring that accurate metrics were obtained following conversions.

Technologies Used

Python

Project Duration

Two Days

Outcomes Achieved

The program was successfully debugged and tested for cross-system compatibility before publishing

Code


        import psutil
        import logging
        logging.basicConfig(filename='system_check.log', level=logging.ERROR)
        def check_cpu():
            try:
                cpu_usage = psutil.cpu_percent()
                return f"CPU Usage: {cpu_usage}%"
            except Exception as e:
                logging.error(f"Error checking CPU: {e}")
                return "Error checking CPU"
        def check_ram():
            try:
                ram = psutil.virtual_memory()
                total_ram = ram.total >> 30 
                used_ram = ram.used >> 30
                return f"RAM Usage: {used_ram:.2f}GB / {total_ram:.2f}GB"
            except Exception as e:
                logging.error(f"Error checking RAM: {e}")
                return "Error checking RAM"
        def check_disk():
            try:
                disk_usage = psutil.disk_usage('/')
                total_disk = disk_usage.total >> 30  
                used_disk = disk_usage.used >> 30
                return f"Disk Usage: {used_disk:.2f}GB / {total_disk:.2f}GB"
            except Exception as e:
                logging.error(f"Error checking HDD: {e}")
                return "Error checking HDD"
        def check_ssd():
            try:
                ssd_usage = psutil.disk_usage('/mnt')  
                total_ssd = ssd_usage.total >> 30  
                used_ssd = ssd_usage.used >> 30
                return f"SSD Usage: {used_ssd:.2f}GB / {total_ssd:.2f}GB"
            except Exception as e:
                logging.error(f"Error checking SSD: {e}")
                return "Error checking SSD"
        def check_disk_io():
            try:
                disk_io = psutil.disk_io_counters()
                return f"Disk I/O: Read Count: {disk_io.read_count}, Write Count: {disk_io.write_count}"
            except Exception as e:
                logging.error(f"Error checking Disk I/O: {e}")
                return "Error checking Disk I/O"
        def check_network_io():
            try:
                net_io = psutil.net_io_counters()
                return f"Network I/O: Bytes Sent: {net_io.bytes_sent}, Bytes Received: {net_io.bytes_recv}"
            except Exception as e:
                logging.error(f"Error checking Network I/O: {e}")
                return "Error checking Network I/O"
        def check_cpu_load():
            try:
                cpu_load = psutil.getloadavg()
                return f"CPU Load Average: {', '.join(map(str, cpu_load))}"
            except Exception as e:
                logging.error(f"Error checking CPU Load Average: {e}")
                return "Error checking CPU Load Average"
        def main():
            try:
                print("Running system checks...")
                print(check_cpu())
                print(check_ram())
                print(check_disk())
                print(check_ssd())
                print(check_disk_io())
                print(check_network_io())
                print(check_cpu_load())
            except Exception as e:
                logging.error(f"Script crashed: {e}")
        if __name__ == "__main__":
            main()
